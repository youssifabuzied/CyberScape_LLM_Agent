Phase 1:
Drone.move_to_point((100, 200, 50))

Drone.communicate_with_apm((100, 200, 50))

Phase 2:
Drone.fly(30)
Drone.move_forward(20)
Drone.detect_with_camera("object")
if Drone.detect_with_camera("object")[1]:
    Drone.communicate_with_apm(Drone.detect_with_camera("object")[0])

Phase 3:
Drone.rotate(90)
Drone.move_forward(15)
Drone.scan_area()
Drone.communicate_with_apm(Drone.get_position_data())

Phase 4:
Drone.wait_for_signal()
Drone.return_to_base()
